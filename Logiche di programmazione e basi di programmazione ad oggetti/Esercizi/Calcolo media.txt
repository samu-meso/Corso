#include <iostream>

int main()
{
    float numero = 0; int contatore = 0; float media = 0; float somma = 0;
    std::cout<<"Inserisci un numero (-1 interrompi): ";
    std::cin >> numero;
    
    while(numero > 0) {
        if(numero < 0) {
            break;
        }
        if(numero > 0) {
            contatore++;
            somma+=numero;
            std::cout<<"Inserisci un altro numero (-1 interrompi): ";
            std::cin >> numero;
        }
    }
    
    if(contatore > 0) {
        media = (somma / contatore);
        std::cout << "Media: "<< media << std::endl;
    } else {
        std::cout << "Non sono stati inseriti numeri!" << std::endl;
    }
    

    return 0;
}
